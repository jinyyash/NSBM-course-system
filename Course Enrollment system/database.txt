create table course(
	CID varchar(10) NOT NULL PRIMARY KEY,	
	name varchar(25),	
	subjectPerSem1 INT,
	subjectPerSem2 INT,
	Year int,
	type varchar(20),
	creditsForSem int
); 
create table faculty(
	FID varchar(10) NOT NULL PRIMARY KEY ,	
	name varchar(25),
	startDate varchar(20)
	
); 

create table student(
	SID varchar(10) NOT NULL PRIMARY KEY ,
	CID int NOT NULL,
	FID int NOT NULL,
	nic varchar(15),
	name varchar(25),
	batch varchar(10),
	address varchar(50),
	dob varchar(10),
	FOREIGN KEY (CID) REFERENCES Courses(CID),
	FOREIGN KEY (FID) REFERENCES Facult(FID)
);
create table facultyDetails(
	FDID int NOT NULL PRIMARY KEY AUTO_INCREMENT,	
	FID int NOT NULL,
	CID int NOT NULL,
	intake varchar(20),
	FOREIGN KEY (CID) REFERENCES Courses(CID),
	FOREIGN KEY (FID) REFERENCES Facult(FID)
);
create table Lecturer(
	LID varchar(10) NOT NULL PRIMARY KEY ,
	nic varchar(15),
	name varchar(25),
	qualification varchar(50),
	speciality varchar(30)
	
);
create table subject(
	SSID varchar(10) NOT NULL PRIMARY KEY,
	LID varchar(10) NOT NULL,
	CID varchar(10) NOT NULL,
	name varchar(25),
	semester varchar(10),
	credits int,
	price decimal(10),
	duration varchar(10),
	FOREIGN KEY (CID) REFERENCES course(CID),
	FOREIGN KEY (LID) REFERENCES Lecturer(LID)
	
	
);	

create table Instructor(
	IID varchar(10) NOT NULL PRIMARY KEY,
	nic varchar(15),
	name varchar(25),
	address varchar(25),
	qualification varchar(50),
	tel integer,
	email varchar(100)
	
);
create table InstructorDetails(
	IDID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	IID int NOT NULL,
	SSID int NOT NULL,
	name varchar(25),
	semester varchar(10),
	credits int,
	price decimal(10),
	duration varchar(10),
	FOREIGN KEY (IID) REFERENCES Instructor(IID),
	FOREIGN KEY (SSID) REFERENCES Subject(SSID)
	
);
create table Lab(
	LID varchar(10) NOT NULL PRIMARY KEY,
	name varchar(25),
	type varchar(25),
	numOfO int,
	numOfCom int	
);
create table LabDetails(
	LDID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	LID int NOT NULL,
	IID int NOT NULL,
	SSID int NOT NULL,
	FOREIGN KEY (LID) REFERENCES Lab(LID),
	FOREIGN KEY (IID) REFERENCES Instructor(IID),
	FOREIGN KEY (SSID) REFERENCES Subject(SSID)
	
);
create table Payment(
	PID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	SID int NOT NULL,
	SSID int NOT NULL,
	date varchar(10),
	time varchar(10),
	FOREIGN KEY (SID) REFERENCES Student(SID),
	FOREIGN KEY (SSID) REFERENCES Subject(SSID)
	
);	
create table Exam(
	EID varchar(10) NOT NULL PRIMARY KEY ,
	SSID varchar(10) NOT NULL,
	examType varchar(10),
	date varchar(10),
	Stime varchar(10),
	Etime varchar(10),
	FOREIGN KEY (SSID) REFERENCES Subject(SSID)
	
);
create table Result(
	RID varchar(10) NOT NULL PRIMARY KEY ,
	EID varchar(10) NOT NULL  ,
	SID varchar(10) NOT NULL  ,
	examType varchar(10),
	result Decimal(10,2),
	grade varchar(10),
	FOREIGN KEY (EID) REFERENCES Exam(EID),
	FOREIGN KEY (SID) REFERENCES Sutudent(SID)	
);
create table Student_Sub(
	SSSID varchar(10) NOT NULL PRIMARY KEY ,
	SSID varchar(10) NOT NULL,
	SID varchar(10) NOT NULL,
	FOREIGN KEY (SSID) REFERENCES Subject(SSID),
	FOREIGN KEY (SID) REFERENCES Student(SID)
	
);
